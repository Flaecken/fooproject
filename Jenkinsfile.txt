pipeline {
    agent any
    stages {
        stage('Checkout') {
            git 'https://github.com/Flaecken/fooproject.git'
        }
    }
    stage('Build') {
        steps {
            bat "mvn compile"
        }
    }
    stage('Test') {
        steps {
            bat "mvn test"
        }
        post {
            always {
                junit '**/TEST*.xml'
            }
        }
    }
}
